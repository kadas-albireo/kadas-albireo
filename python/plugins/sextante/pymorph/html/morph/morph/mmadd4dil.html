def mmadd4dil(f, c):
    from Numeric import asarray, minimum, maximum
    if c:            
       y = asarray(f,'d') + c
       k1,k2 = mmlimits(f)
       y = ((f==k1) * k1) + ((f!=k1) * y)
       y = maximum(minimum(y,k2),k1)
       a = y.astype(f.typecode())
    else:
       a = f
    return a
